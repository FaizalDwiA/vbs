Function AddFunct(value1, value2)
    dim total

    if IsNumeric(value1) And IsNumeric(value1) then
        total = CInt(value1) + CInt(value2)
        AddFunct = total
    else
        total = value1 & " " & value2
        AddFunct = total
    End if
end Function
