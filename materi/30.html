For Each Next Loops

Basic For

Option Explicit
dim n

for n = 0 to 4
    MsgBox "Looping"
next

MsgBox "Finished"

Exit For

Option Explicit
dim n

for n = 0 to 4
    if n = 2 then
        Exit For
    end if
    MsgBox "Looping"
next

MsgBox "Finished"

Mengetahui Isi File dari sebuah Folder dengan For Each 

Option Explicit
dim fso, folder, files

Set fso = CreateObject("Scripting.FileSystemObject")
set folder = fso.GetFolder("C:\Users\Program Is Fun\downloads\faizal\")

for Each files in folder.files
    MsgBox files.name
next

Mengetahui Ukuran dari sub Folder 

Option Explicit
dim fso, folder, sf

Set fso = CreateObject("Scripting.FileSystemObject")
set folder = fso.GetFolder("C:\Users\Program Is Fun\downloads\faizal\")

for Each sf in folder.SubFolders
    MsgBox sf.size
next

Mengetahui Nama dan Ukuran Folder 

Option Explicit
dim fso, folder, sf

Set fso = CreateObject("Scripting.FileSystemObject")
set folder = fso.GetFolder("C:\Users\Program Is Fun\downloads\faizal\")

for Each sf in folder.SubFolders
    MsgBox "Name: " & sf.name & vbLf & _
            "Ukuran: " & sf.Size
next
