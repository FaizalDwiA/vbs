Lcase Ucase Trim

Lcase : mengecilkan huruf

L : Lower

Option Explicit
dim a

a = "This is a message. No duh..."

MsgBox LCase(a)

Ucase : Memperbesar Huruf

U : Upper

Option Explicit
dim a

a = "This is a message. No duh..."

MsgBox UCase(a)

Example

Option Explicit
dim a

a = "This is a message. No duh... Cantaloupe."

if InStr(LCase(a), "cantaloupe") then
    MsgBox "yes"
else
    MsgBox "no"
end if
