# File Folder Exits

## Basic

```vbs
CreateObject("scripting.fileSystemObject")
```

## Nilai / Value

```vbs
Option Explicit
dim fso

Set fso = CreateObject("Scripting.FileSystemObject")
MsgBox fso.FileExists("C:\Users\PIF\Desktop\coba.txt")
```

## If Else

```vbs
Option Explicit
dim fso

Set fso = CreateObject("Scripting.FileSystemObject")

if fso.FileExists("C:\Users\PIF\Desktop\coba.txt") Then
    WScript.echo "yes"
else
    WScript.echo "no"
end if
```

## Folder Exits

```vbs
Option Explicit
dim fso

Set fso = CreateObject("Scripting.FileSystemObject")

if fso.FolderExists("C:\Users\PIF\Desktop\PIF") Then
    WScript.echo "yes"
else
    WScript.echo "no"
end if
```

## If not

```vbs
Option Explicit
dim fso

Set fso = CreateObject("Scripting.FileSystemObject")

if not fso.FolderExists("C:\Users\PIF\Desktop\PIF") Then
    WScript.echo "yes"
else
    WScript.echo "no"
end if
```
