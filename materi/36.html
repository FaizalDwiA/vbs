Option Explicit
dim pass

do
    pass = InputBox("Please type your password:", "Verification")
    if TypeName(pass) = "Empty" then
        WScript.Quit
    ElseIf pass = "" Then
        MsgBox "Please do not leave field blank."
    ElseIf pass = "syiber" Then
        MsgBox "Correct"
        exit do
    else
        MsgBox "Please enter the correct password"
    end if
loop
