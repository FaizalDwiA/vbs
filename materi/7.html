Do Until

untuk berhenti dengan syarat

Option Explicit
dim n

n = 1
do Until n = 5
    MsgBox n
    n = n + 1
Loop

Until berada di akhir

Option Explicit
dim n

n = 1
do
    MsgBox n
    n = n + 1
Loop Until n = 5

Menggunakan lebih dari

Option Explicit
dim n

n = 1
do Until n > 5
    n = n + 1
    MsgBox n
Loop

Do While

Option Explicit
dim n

n = 0
do while n < 5
    n = n + 1
    MsgBox n
Loop

bisa juga menggunakan <>

Option Explicit

bisa juga seperti ini:

Option Explicit
dim pass

do
    pass = InputBox("Password: ")
