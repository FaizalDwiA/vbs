Automation: Internet Explorer

Example

option explicit
dim ie, x

set ie = createobject("InternetExplorer.Application")
set x = createobject("wscript.shell")

ie.Navigate "http://google.com/"
ie.Visible=1
ie.Toolbar=0
ie.StatusBar=0
ie.Height=560
ie.Width=1000
ie.Top=0
ie.Left=0
ie.Resizeble=0

x.sendkeys "cow"
x.sendkeys "{enter}"

Example Facebook

option explicit
dim ie, x

set ie = createobject("InternetExplorer.Application")
set x = createobject("wscript.shell")

sub waitforload
    Do While ie.Busy
        wscript.sleep 200
    Loop
end sub


ie.Navigate "http://www.facebook.com/"
ie.Visible=1
ie.Toolbar=0
ie.StatusBar=0
ie.Height=560
ie.Width=1000
ie.Top=0
ie.Left=0
ie.Resizeble=0

call waitforload
x.sendkeys "cow"
x.sendkeys "{tab}"
x.sendkeys "pass"
